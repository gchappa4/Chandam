    var l = document.getElementsByTagName("td");
    for (var i = 0; i < l.length; i++) {
    var e = l.item(i);
    if (e.getAttribute("nowrap") != null) {
    MiriyamAPI.Worker.beautify(e);
    }
    }
    }, 0);
    });
